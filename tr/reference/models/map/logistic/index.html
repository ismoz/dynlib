    logistic.toml

    Kaynak: src/dynlib/models/map/logistic.toml 

    [model]
type="map"
name="Logistic Map"

[states]
x=0.1

[params]
r=4.0

[equations.rhs]
x = "r * x * (1 - x)"

[equations.jacobian]
expr = [
    ["r - 2 * r * x"]
]
